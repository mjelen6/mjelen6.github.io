180 /*----------------------------------------------------------------------------
181  Every 1 ms function that makes FSM run
182  *----------------------------------------------------------------------------*/
183 void ledsService1ms (void) {
184 
185  if(nextStateDelay!=PHASE_DELAY_MS){
186  nextStateDelay+=startFSM;
187  return;
188  }
189  nextStateDelay=0;
190 
191  nextLedState();
192 
193 }

195 /*----------------------------------------------------------------------------
196  Function that makes FSM transition to the next state
197  *----------------------------------------------------------------------------*/
198 void nextLedState (void) {
199 
201  if (slowFSM==0) {
202  skipPhase+=1;
203  if( (skipPhase%3) != 0 ) return; /* Every third call passes to switch clause*/
204  }
205 
206 
207  switch(phase){
208  case PHASE_A:
209  phase=phaseA();
210  break;
211  case PHASE_B:
212  phase=phaseB();
213  break;
214  default:
215  phase=phaseC();
216  break;
217  }
218 }

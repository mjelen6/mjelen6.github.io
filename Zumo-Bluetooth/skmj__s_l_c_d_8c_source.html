218 void sLCD_setByType(uint32_t value, char type){
219 // type ( 'd' value (0...9999) - decimal
220 // 'b' value (0...15) - binary
221 // 'h' value (0...65535) - hexadecimal
222 // 'o' value (0...4095) ) - octal
223 
224  if( (type == 'd') && (value >= 0) && (value < 10000) ){
225 
226  // konwersja na system dziesietny i wyswietlanie
227  sLCD_set(value/1000,1);
228  sLCD_set((value/100)%10,2);
229  sLCD_set((value/10)%10,3);
230  sLCD_set((value%10),4);
231  }
232  else if( (type == 'b') && (value >=0) && (value < 16) ){
233  uint8_t i=0;
234  uint8_t x;
235 
236  // konwersja na system dwojkowy i wyswietlanie
237  for( i=0; i<4; ++i){
238 
239  x = (value >> (3-i)) & 0x01;
240  sLCD_set( x,i+1);
241  }
242  }
243  else if(type == 'h'){
244  uint8_t i=0;
245  uint16_t x;
246 
247  // konwersja na system szesnastkowy i wyswietlanie
248  for( i=0; i<4; ++i){
249 
250  x = (value >> (12-4*i)) & 0x0f;
251  sLCD_set( x,i+1);
252  }
253  }
254  else if( (type == 'o') && (value < 4096) && (value >= 0) ){
255  uint8_t i=0;
256  uint16_t x;
257 
258  // konwersja na system osemkowy i wyswietlanie
259  for( i=0; i<4; ++i){
260 
261  x = (value >> (9-3*i)) & 0x07;
262  sLCD_set( x,i+1);
263  }
264  }
265  else{
266  sLCD_error();
267  }
268 }

270 void sLCD_error(void){
271 
272  LCD->WF8B[LCD_Front_Pin[((2*1)-2)]] = 0;
273  LCD->WF8B[LCD_Front_Pin[((2*1)-1)]] = 0;
274 
275  // 'E'
276  LCD->WF8B[LCD_Front_Pin[((2*2)-2)]] = LCD_Digit_2[14];
277  LCD->WF8B[LCD_Front_Pin[((2*2)-1)]] = LCD_Digit_1[14];
278 
279  // 'r'
280  LCD->WF8B[LCD_Front_Pin[((2*3)-2)]] = LCD_Digit_2[16];
281  LCD->WF8B[LCD_Front_Pin[((2*3)-1)]] = LCD_Digit_1[16];
282 
283  // 'r'
284  LCD->WF8B[LCD_Front_Pin[((2*4)-2)]] = LCD_Digit_2[16];
285  LCD->WF8B[LCD_Front_Pin[((2*4)-1)]] = LCD_Digit_1[16];
286 }
